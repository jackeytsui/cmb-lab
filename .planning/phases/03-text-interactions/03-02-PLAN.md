```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@clerk/nextjs/server";
import type { GradingRequest, GradingResponse } from "@/lib/grading";

export async function POST(request: NextRequest) {
  // 1. Verify user is authenticated
  const { userId } = await auth();
  if (!userId) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  try {
    const body = await request.json();

    // 2. Validate request body
    const { interactionId, studentResponse, expectedAnswer, language } = body;
    if (!interactionId || !studentResponse) {
      return NextResponse.json(
        { error: "Missing required fields: interactionId, studentResponse" },
        { status: 400 }
      );
    }

    // 3. Check if n8n webhook URL is configured
    const webhookUrl = process.env.N8N_GRADING_WEBHOOK_URL;
    if (!webhookUrl) {
      // Development fallback: return consistent mock response
      // This allows UI testing without n8n, but does NOT test real Chinese grading
      console.warn(
        "N8N_GRADING_WEBHOOK_URL not configured. Returning mock response. " +
        "Configure webhook for real AI grading of Chinese responses."
      );
      const mockResponse: GradingResponse = {
        isCorrect: true,
        score: 85,
        feedback: "Mock response: Your answer has been accepted. Configure N8N_GRADING_WEBHOOK_URL for real AI grading.",
        corrections: undefined,
        hints: undefined,
      };
      return NextResponse.json(mockResponse);
    }

    // 4. Build n8n request payload
    const gradingRequest: GradingRequest = {
      interactionId,
      userId,
      studentResponse,
      expectedAnswer,
      language: language || "both",
    };

    // 5. Call n8n webhook with 15 second timeout
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), 15000);

    const n8nResponse = await fetch(webhookUrl, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        ...(process.env.N8N_WEBHOOK_AUTH_HEADER && {
          Authorization: process.env.N8N_WEBHOOK_AUTH_HEADER,
        }),
      },
      body: JSON.stringify(gradingRequest),
      signal: controller.signal,
    });

    clearTimeout(timeoutId);

    if (!n8nResponse.ok) {
      console.error(`n8n webhook failed: ${n8nResponse.status}`);
      return NextResponse.json(
        { error: "Grading service unavailable" },
        { status: 502 }
      );
    }

    // 6. Parse and return grading result
    const gradingResult: GradingResponse = await n8nResponse.json();
    return NextResponse.json(gradingResult);

  } catch (error) {
    if (error instanceof Error && error.name === "AbortError") {
      return NextResponse.json(
        { error: "Grading request timed out" },
        { status: 504 }
      );
    }
    console.error("Grading API error:", error);
    return NextResponse.json(
      { error: "Failed to grade response" },
      { status: 500 }
    );
  }
}
```

Key points:
- Auth check using Clerk
- Request validation
- Development fallback when N8N_GRADING_WEBHOOK_URL not set returns consistent mock (score 85, always passes)
- Mock response clearly indicates it's a mock and prompts to configure webhook
- 15 second timeout for n8n call
- Proper error handling for timeout, n8n failure, general errors

```typescript
"use client";

import { CheckCircle, XCircle, Lightbulb } from "lucide-react";
import { motion } from "framer-motion";
import type { GradingFeedback } from "@/lib/grading";

interface FeedbackDisplayProps {
  feedback: GradingFeedback;
}

export function FeedbackDisplay({ feedback }: FeedbackDisplayProps) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 10 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -10 }}
      className={`p-4 rounded-lg ${
        feedback.isCorrect
          ? "bg-green-500/20 border border-green-500/50"
          : "bg-red-500/20 border border-red-500/50"
      }`}
    >
      <div className="flex items-start gap-3">
        {feedback.isCorrect ? (
          <CheckCircle className="h-6 w-6 text-green-500 shrink-0" />
        ) : (
          <XCircle className="h-6 w-6 text-red-500 shrink-0" />
        )}
        <div className="space-y-2 flex-1">
          {/* Main feedback message */}
          <p className="text-white font-medium">{feedback.feedback}</p>

          {/* Score badge */}
          <div className="flex items-center gap-2">
            <span
              className={`text-sm px-2 py-0.5 rounded ${
                feedback.isCorrect
                  ? "bg-green-500/30 text-green-300"
                  : "bg-red-500/30 text-red-300"
              }`}
            >
              Score: {feedback.score}/100
            </span>
          </div>

          {/* Corrections list */}
          {feedback.corrections && feedback.corrections.length > 0 && (
            <div className="text-sm text-zinc-300 mt-2">
              <p className="font-medium text-zinc-200">Corrections:</p>
              <ul className="list-disc list-inside mt-1 space-y-1">
                {feedback.corrections.map((correction, index) => (
                  <li key={index}>{correction}</li>
                ))}
              </ul>
            </div>
          )}

          {/* Hints for incorrect answers */}
          {!feedback.isCorrect &&
            feedback.hints &&
            feedback.hints.length > 0 && (
              <div className="flex items-start gap-2 text-sm text-yellow-400 mt-2 p-2 bg-yellow-500/10 rounded">
                <Lightbulb className="h-4 w-4 shrink-0 mt-0.5" />
                <div>
                  <p className="font-medium">Hint:</p>
                  <p>{feedback.hints[0]}</p>
                </div>
              </div>
            )}
        </div>
      </div>
    </motion.div>
  );
}
```

Key points:
- Uses Framer Motion for enter/exit animations
- Green styling for correct, red for incorrect
- Displays score badge
- Shows corrections list if provided
- Shows first hint in highlighted box for incorrect answers
- Uses lucide-react icons (already in project)

```typescript
async function handleSubmit(values: TextInteractionFormValues) {
  setIsSubmitting(true);
  setFeedback(null); // Clear previous feedback

  try {
    let result: GradingFeedback;

    if (onSubmit) {
      // Use provided submit handler (for testing/custom logic)
      result = await onSubmit(values.response);
    } else {
      // Call the grading API
      const response = await fetch("/api/grade", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          interactionId,
          studentResponse: values.response,
          expectedAnswer,
          language,
        }),
      });

      if (!response.ok) {
        throw new Error(`Grading failed: ${response.status}`);
      }

      result = await response.json();
    }

    setFeedback(result);

    if (result.isCorrect) {
      // Show success feedback briefly before completing
      setTimeout(() => {
        onComplete();
      }, 1500);
    }
  } catch (error) {
    console.error("Grading error:", error);
    setFeedback({
      isCorrect: false,
      score: 0,
      feedback: "An error occurred while grading. Please try again.",
    });
  } finally {
    setIsSubmitting(false);
  }
}
```
