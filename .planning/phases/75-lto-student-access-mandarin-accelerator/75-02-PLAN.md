       ```typescript
       function normalizeForComparison(text: string): string {
         return text.normalize("NFC")
           .replace(/[\u200B\u200C\u200D\uFEFF\uFFFD]/g, "")
           .replace(/[\s\u3000]+/g, "")
           .replace(/[，。！？、；：""''（）《》【】…—·\.\,\!\?\;\:\"\'\(\)\<\>\[\]]/g, "")
           .trim();
       }
       ```
     - Compare: `normalizeForComparison(input) === normalizeForComparison(expected)`
     - If CORRECT: show green background on input, show a checkmark icon, play a brief success animation (CSS only, green pulse), then auto-advance to next sentence after 800ms. POST to `/api/accelerator/typing/progress` with the sentenceId.

   - Character-by-character feedback on wrong answer (visual only, shown below input):
     ```typescript
     function getCharFeedback(input: string, expected: string): Array<{ char: string; correct: boolean }> {
       const normInput = normalizeForComparison(input);
       const normExpected = normalizeForComparison(expected);
       return [...normInput].map((char, i) => ({
         char,
         correct: i < normExpected.length && char === normExpected[i],
       }));
     }
     ```
