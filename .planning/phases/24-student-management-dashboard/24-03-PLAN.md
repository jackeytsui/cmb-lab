export default async function AdminStudentsPage({
  searchParams,
}: {
  searchParams: Promise<Record<string, string | string[]>>;
}) {
  const params = await searchParams;
  const page = Number(params.page) || 1;
  const pageSize = Number(params.pageSize) || 25;
  const sortBy = (params.sortBy as string) || "createdAt";
  const sortOrder = ((params.sortOrder as string) || "desc") as "asc" | "desc";
  const search = (params.search as string) || "";
  const tagIds = params.tagIds ? String(params.tagIds).split(",").filter(Boolean) : [];
  const courseId = (params.courseId as string) || undefined;
  const atRisk = params.atRisk === "true";
