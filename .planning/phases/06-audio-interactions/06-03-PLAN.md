**Type Strategy Clarification:**

The `InteractionCuePoint` interface in `src/lib/interactions.ts` already extends `CuePoint` with the fields needed:
- `language: LanguagePreference`
- `type?: "text" | "audio"`
- `prompt?: string`
- `expectedAnswer?: string`

Do NOT modify `src/types/video.ts` (CuePoint). Use `InteractionCuePoint` throughout.

**Update `src/components/video/InteractiveVideoPlayer.tsx`:**

1. Import AudioInteraction and TextInteraction:
   ```typescript
   import { AudioInteraction } from "@/components/audio/AudioInteraction"
   import { TextInteraction } from "@/components/interactions/TextInteraction"
   ```

2. Add state for active interaction:
   ```typescript
   const [activeInteraction, setActiveInteraction] = useState<InteractionCuePoint | null>(null)
   ```

3. Update handleCuePointChange to store active cue point:
   In the existing handleCuePointChange callback, after determining activeCuePoint:
   ```typescript
   const handleCuePointChange = useCallback(
     (event: Event) => {
       const player = event.target as HTMLVideoElement & {
         activeCuePoint?: { value?: { cuePoint?: CuePoint } };
       };

       const activeCuePoint = player.activeCuePoint?.value?.cuePoint;
       if (activeCuePoint && !activeCuePoint.completed) {
         // Cast to InteractionCuePoint for type/prompt/etc fields
         const interactionCuePoint = activeCuePoint as InteractionCuePoint;
         setActiveInteraction(interactionCuePoint);
         handleCuePointReached(activeCuePoint);
         onInteractionRequired(activeCuePoint);
       }
     },
     [handleCuePointReached, onInteractionRequired]
   );
   ```

4. Create interaction completion handler:
   ```typescript
   const handleInteractionDone = useCallback(() => {
     if (activeInteraction) {
       // Mark cue point completed
       const updatedCuePoints = context.cuePoints.map(cp =>
         cp.id === activeInteraction.id ? { ...cp, completed: true } : cp
       );
       setCuePoints(updatedCuePoints);
       setActiveInteraction(null);
     }
     completeInteraction();
     onInteractionComplete?.();
     markInteractionComplete();
   }, [activeInteraction, context.cuePoints, setCuePoints, completeInteraction, onInteractionComplete, markInteractionComplete]);
   ```

5. Render appropriate interaction in overlay:
   Replace the existing InteractionOverlay children logic with:
   ```typescript
   <InteractionOverlay
     isVisible={isInteractionPending}
     onExitComplete={handleOverlayExitComplete}
     sidebarContent={sidebarContent}
   >
     {children || (activeInteraction ? (
       activeInteraction.type === 'audio' ? (
         <AudioInteraction
           interactionId={activeInteraction.interactionId}
           lessonId={lessonId || ''}
           prompt={activeInteraction.prompt || 'Record your response'}
           expectedAnswer={activeInteraction.expectedAnswer}
           language={activeInteraction.language || 'both'}
           onComplete={handleInteractionDone}
         />
       ) : (
         <TextInteraction
           interactionId={activeInteraction.interactionId}
           prompt={activeInteraction.prompt || 'Type your response'}
           expectedAnswer={activeInteraction.expectedAnswer}
           language={activeInteraction.language || 'both'}
           onComplete={handleInteractionDone}
         />
       )
     ) : (
       <div className="text-center text-white">
         <p className="text-lg">Interaction content will appear here</p>
       </div>
     ))}
   </InteractionOverlay>
   ```

**Key implementation notes:**
- The `children` prop still takes precedence for backwards compatibility
- Default to 'text' type if `type` field is not specified (backwards compatible)
- Use `activeInteraction.interactionId` (not `activeInteraction.id`) for the interaction component
- Clear `activeInteraction` state when interaction completes
