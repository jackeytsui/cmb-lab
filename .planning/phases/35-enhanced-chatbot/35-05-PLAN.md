**Add PhoneticText to user-typed Chinese text (CHAT-02):**

Create a `renderUserText` function that detects Chinese character segments and wraps them in PhoneticText:

```typescript
/**
 * Renders user-typed text with PhoneticText applied to Chinese character segments.
 * English text renders in default font; Chinese text gets phonetic annotations.
 */
function renderUserText(text: string) {
  // Regex to match runs of Chinese characters (CJK Unified Ideographs + Extension A)
  const chineseRegex = /([\u4e00-\u9fff\u3400-\u4dbf]+)/g;
  const segments = text.split(chineseRegex);

  return segments.map((segment, i) => {
    if (chineseRegex.test(segment)) {
      // Reset regex lastIndex after test
      chineseRegex.lastIndex = 0;
      return <PhoneticText key={i}>{segment}</PhoneticText>;
    }
    return <span key={i}>{segment}</span>;
  });
}
```

Note: Use a NEW regex instance or reset lastIndex because the `g` flag maintains state. Better approach: use a non-global regex for the test:

```typescript
const CHINESE_CHAR_REGEX = /[\u4e00-\u9fff\u3400-\u4dbf]/;

function renderUserText(text: string) {
  const splitRegex = /([\u4e00-\u9fff\u3400-\u4dbf]+)/g;
  const segments = text.split(splitRegex);

  return segments.map((segment, i) => {
    if (CHINESE_CHAR_REGEX.test(segment)) {
      return <PhoneticText key={i}>{segment}</PhoneticText>;
    }
    return <span key={i}>{segment}</span>;
  });
}
```

This keeps the existing `renderAnnotatedText` for ASSISTANT messages (which use `[char|pinyin|jyutping]` annotation format) and adds `renderUserText` for USER messages (which are plain text that may contain Chinese characters).
