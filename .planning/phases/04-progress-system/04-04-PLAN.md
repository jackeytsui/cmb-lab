   const handleTimeUpdate = useCallback(
     (event: Event) => {
       const player = event.target as HTMLVideoElement;
       const currentTime = player.currentTime;
       const duration = player.duration;

       updateTime(currentTime);

       // Track progress if duration is valid
       if (duration && duration !== Infinity) {
         const percent = Math.floor((currentTime / duration) * 100);
         // Only update every 5% to reduce API calls
         const lastPercent = Math.floor((context.currentTime / duration) * 100);
         if (percent >= lastPercent + 5 || percent >= 95) {
           updateVideoProgress(percent);
         }
       }
     },
     [updateTime, updateVideoProgress, context.currentTime]
   );
